111 /**
112  * X.509 Certificate class to sign and generate hex encoded certificate
113  * @name KJUR.asn1.x509.Certificate
114  * @class X.509 Certificate class to sign and generate hex encoded certificate
115  * @param {Array} params associative array of parameters (ex. {'tbscertobj': obj, 'rsaprvkey': key})
116  * @extends KJUR.asn1.ASN1Object
117  * @description
118  * <br/>
119  * As for argument 'params' for constructor, you can specify one of
120  * following properties:
121  * <ul>
122  * <li>tbscertobj - specify {@link KJUR.asn1.x509.TBSCertificate} object</li>
123  * <li>rsaprvkey - specify {@link RSAKey} object CA private key</li>
124  * </ul>
125  * NOTE: 'params' can be omitted.
126  * <h4>EXAMPLE</h4>
127  * @example
128  * var prvKey = new RSAKey(); // CA's private key
129  * prvKey.readPrivateKeyFromASN1HexString("3080...");
130  * var cert = new KJUR.asn1x509.Certificate({'tbscertobj': tbs, 'rsaprvkey': prvKey});
131  * cert.sign(); // issue certificate by CA's private key
132  * var hCert = cert.getEncodedHex();
133  *
134  * // Certificate  ::=  SEQUENCE  {
135  * //     tbsCertificate       TBSCertificate,
136  * //     signatureAlgorithm   AlgorithmIdentifier,
137  * //     signature            BIT STRING  }	    
138  */
139 KJUR.asn1.x509.Certificate = function(params) {
140     KJUR.asn1.x509.Certificate.superclass.constructor.call(this);
141     var asn1TBSCert = null;
142     var asn1SignatureAlg = null;
143     var asn1Sig = null;
144     var hexSig = null;
145     var rsaPrvKey = null;
146 
147     /**
148      * sign TBSCertificate and set signature value internally
149      * @name sign
150      * @memberOf KJUR.asn1.x509.Certificate
151      * @function
152      * @description
153      * @example
154      * var cert = new KJUR.asn1.x509.Certificate({'tbscertobj': tbs, 'rsaprvkey': prvKey});
155      * cert.sign();
156      */
157     this.sign = function() {
158 	this.asn1SignatureAlg = this.asn1TBSCert.asn1SignatureAlg;
159 
160 	sig = new KJUR.crypto.Signature({'alg': 'SHA1withRSA', 'prov': 'cryptojs/jsrsa'});
161 	sig.initSign(this.rsaPrvKey);
162 	sig.updateHex(this.asn1TBSCert.getEncodedHex());
163 	this.hexSig = sig.sign();
164 
165 	this.asn1Sig = new KJUR.asn1.DERBitString({'hex': '00' + this.hexSig});
166 	
167 	var seq = new KJUR.asn1.DERSequence({'array': [this.asn1TBSCert,
168 						       this.asn1SignatureAlg,
169 						       this.asn1Sig]});
170 	this.hTLV = seq.getEncodedHex();
171 	this.isModified = false;
172     };
173 
174     this.getEncodedHex = function() {
175 	if (this.isModified == false && this.hTLV != null) return this.hTLV;
176 	throw "not signed yet";
177     };
178 
179     /**
180      * get PEM formatted certificate string after signed
181      * @name getPEMString
182      * @memberOf KJUR.asn1.x509.Certificate
183      * @function
184      * @return PEM formatted string of certificate
185      * @description
186      * @example
187      * var cert = new KJUR.asn1.x509.Certificate({'tbscertobj': tbs, 'rsaprvkey': prvKey});
188      * cert.sign();
189      * var sPEM =  cert.getPEMString();
190      */
191     this.getPEMString = function() {
192 	var hCert = this.getEncodedHex();
193 	var wCert = CryptoJS.enc.Hex.parse(hCert);
194 	var b64Cert = CryptoJS.enc.Base64.stringify(wCert);
195 	var pemBody = b64Cert.replace(/(.{64})/g, "$1\r\n");
196 	return "-----BEGIN CERTIFICATE-----\r\n" + pemBody + "\r\n-----END CERTIFICATE-----\r\n";
197     };
198 
199     if (typeof params != "undefined") {
200 	if (typeof params['tbscertobj'] != "undefined") {
201 	    this.asn1TBSCert = params['tbscertobj'];
202 	}
203 	if (typeof params['rsaprvkey'] != "undefind") {
204 	    this.rsaPrvKey = params['rsaprvkey'];
205 	}
206     }
207 };

293      * set notBefore field by parameter
294      * @name setNotBeforeByParam
295      * @memberOf KJUR.asn1.x509.TBSCertificate

299      * @example
300      * tbsc.setNotBeforeByParam({'str': '130508235959Z'});

353      * append X.509v3 extension to this object

368     this.getEncodedHex = function() {
369 	if (this.asn1NotBefore == null || this.asn1NotAfter == null)
370 	    throw "notBefore and/or notAfter not set";
371 	var asn1Validity = 
372 	    new KJUR.asn1.DERSequence({'array':[this.asn1NotBefore, this.asn1NotAfter]});

374 	this.asn1Array = new Array();
375 
376 	this.asn1Array.push(this.asn1Version);
377 	this.asn1Array.push(this.asn1SerialNumber);

381 	this.asn1Array.push(this.asn1Subject);
382 	this.asn1Array.push(this.asn1SubjPKey);
383 
384 	if (this.extensionsArray.length > 0) {
385 	    var extSeq = new KJUR.asn1.DERSequence({"array": this.extensionsArray});
386 	    var extTagObj = new KJUR.asn1.DERTaggedObject({'explicit': true,
387 							   'tag': 'a3',
388 							   'obj': extSeq});
389 	    this.asn1Array.push(extTagObj);
390 	}
391 
392 	var o = new KJUR.asn1.DERSequence({"array": this.asn1Array});
393 	this.hTLV = o.getEncodedHex();
394 	this.isModified = false;
395 	return this.hTLV;
396     };
397 
398     this._initialize();
399 };

406 /**
407  * base Extension ASN.1 structure class
408  * @name KJUR.asn1.x509.Extension
409  * @class base Extension ASN.1 structure class
410  * @param {Array} params associative array of parameters (ex. {'critical': true})
411  * @extends KJUR.asn1.ASN1Object
412  * @description
413  * @example
414  * // Extension  ::=  SEQUENCE  {
415  * //     extnID      OBJECT IDENTIFIER,
416  * //     critical    BOOLEAN DEFAULT FALSE,
417  * //     extnValue   OCTET STRING  }
418  */
419 KJUR.asn1.x509.Extension = function(params) {
420     KJUR.asn1.x509.Extension.superclass.constructor.call(this);
421     var asn1ExtnValue = null;
422 
423     this.getEncodedHex = function() {
424 	var asn1Oid = new KJUR.asn1.DERObjectIdentifier({'oid': this.oid});
425 	var asn1EncapExtnValue = 
426 	    new KJUR.asn1.DEROctetString({'hex': this.getExtnValueHex()});
427 
428 	var asn1Array = new Array();
429 	asn1Array.push(asn1Oid);
430 	if (this.critical) asn1Array.push(new KJUR.asn1.DERBoolean());
431 	asn1Array.push(asn1EncapExtnValue);
432 
433 	var asn1Seq = new KJUR.asn1.DERSequence({'array': asn1Array});
434 	return asn1Seq.getEncodedHex();
435     };
436 
437     this.critical = false;
438     if (typeof params != "undefined") {
439 	if (typeof params['critical'] != "undefined") {
440 	    this.critical = params['critical'];
441 	}
442     }
443 };
