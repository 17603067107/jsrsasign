 37 /**
 38  * static object for cryptographic function utilities
 39  * @name KJUR.crypto.Util
 40  * @class static object for cryptographic function utilities
 41  * @property {Array} DIGESTINFOHEAD PKCS#1 DigestInfo heading hexadecimal bytes for each hash algorithms
 42  * @description
 43  */
 44 KJUR.crypto.Util = new function() {
 45     this.DIGESTINFOHEAD = {
 46 	'sha1':      "3021300906052b0e03021a05000414",
 47         'sha224':    "302d300d06096086480165030402040500041c",
 48 	'sha256':    "3031300d060960864801650304020105000420",
 49 	'sha384':    "3041300d060960864801650304020205000430",
 50 	'sha512':    "3051300d060960864801650304020305000440",
 51 	'md2':       "3020300c06082a864886f70d020205000410",
 52 	'md5':       "3020300c06082a864886f70d020505000410",
 53 	'ripemd160': "3021300906052b2403020105000414"
 54     };
 55 
 56     /**
 57      * get hexadecimal DigestInfo
 58      * @name getDigestInfoHex
 59      * @memberOf KJUR.crypto.Util
 60      * @function
 61      * @param {String} hHash hexadecimal hash value
 62      * @param {String} alg hash algorithm name (ex. 'sha1')
 63      * @return {String} hexadecimal string DigestInfo ASN.1 structure
 64      */
 65     this.getDigestInfoHex = function(hHash, alg) {
 66 	if (typeof this.DIGESTINFOHEAD[alg] == "undefined")
 67 	    throw "alg not supported in Util.DIGESTINFOHEAD: " + alg;
 68 	return this.DIGESTINFOHEAD[alg] + hHash;
 69     };
 70 
 71     /**
 72      * get PKCS#1 padded hexadecimal DigestInfo
 73      * @name getPaddedDigestInfoHex
 74      * @memberOf KJUR.crypto.Util
 75      * @function
 76      * @param {String} hHash hexadecimal hash value
 77      * @param {String} alg hash algorithm name (ex. 'sha1')
 78      * @param {Integer} keySize key bit length (ex. 1024)
 79      * @return {String} hexadecimal string of PKCS#1 padded DigestInfo
 80      */
 81     this.getPaddedDigestInfoHex = function(hHash, alg, keySize) {
 82 	var hDigestInfo = this.getDigestInfoHex(hHash, alg);
 83 	var pmStrLen = keySize / 4; // minimum PM length
 84 
 85 	if (hDigestInfo.length + 22 > pmStrLen) // len(0001+ff(*8)+00+hDigestInfo)=22
 86 	    throw "key is too short for SigAlg: keylen=" + keySize + "," + alg;
 87 
 88 	var hHead = "0001";
 89 	var hTail = "00" + hDigestInfo;
 90 	var hMid = "";
 91 	var fLen = pmStrLen - hHead.length - hTail.length;
 92 	for (var i = 0; i < fLen; i += 2) {
 93 	    hMid += "ff";
 94 	}
 95 	var hPaddedMessage = hHead + hMid + hTail;
 96 	return hPaddedMessage;
 97     };
 98 };

100 /**
101  * MessageDigest class which is very similar to java.security.MessageDigest class
102  * @name KJUR.crypto.MessageDigest
103  * @class MessageDigest class which is very similar to java.security.MessageDigest class
104  * @param {Array} params parameters for constructor
105  * @description
106  * <br/>
107  * Currently this supports md5, sha1, sha224, sha256, sha384,
108  * sha512 and ripemd160 for algorithm and 'cryptojs' for provider.
109  * @example
110  * var md = new KJUR.crypto.MessageDigest({"alg": "sha1", "prov": "cryptojs"});
111  * // append data
112  * md.updateHex('1f2d3e')
113  * md.updateString('aaa')
114  * // get message digest
115  * md.digest()
116  * md.digestHex('5f6de0')
117  * md.digestString('aaa')
118  * @see http://crypto-js.googlecode.com/svn/tags/3.1.2/build/components/core-min.js
119  * @see http://crypto-js.googlecode.com/svn/tags/3.1.2/build/rollups/sha1.js
120  */
121 KJUR.crypto.MessageDigest = function(params) {
122     var md = null;
123     var algName = null;
124     var provName = null;
125     var _CryptoJSMdName = {
126 	'md5': 'CryptoJS.algo.MD5',
127 	'sha1': 'CryptoJS.algo.SHA1',
128 	'sha224': 'CryptoJS.algo.SHA224',
129 	'sha256': 'CryptoJS.algo.SHA256',
130 	'sha384': 'CryptoJS.algo.SHA384',
131 	'sha512': 'CryptoJS.algo.SHA512',
132 	'ripemd160': 'CryptoJS.algo.RIPEMD160'
133     };
134 
135     /**
136      * set hash algorithm and provider
137      * @name setAlgAndProvider
138      * @memberOf KJUR.crypto.MessageDigest
139      * @function
140      * @param {String} alg hash algorithm name
141      * @param {String} prov provider name
142      * @description
143      * <br/>
144      * <h4>EXAMPLES</h4>
145      * @example
146      * // for SHA1
147      * md.setAlgAndProvider('sha1', 'cryptojs');
148      * // for RIPEMD160
149      * md.setAlgAndProvider('ripemd160', 'cryptojs');
150      */
151     this.setAlgAndProvider = function(alg, prov) {
152 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(alg) != -1 &&
153 	    prov == 'cryptojs') {
154 	    try {
155 		this.md = eval(_CryptoJSMdName[alg]).create();
156 	    } catch (ex) {
157 		throw "setAlgAndProvider hash alg set fail alg=" + alg + "/" + ex;
158 	    }
159 	    this.updateString = function(str) {
160 		this.md.update(str);
161 	    };
162 	    this.updateHex = function(hex) {
163 		var wHex = CryptoJS.enc.Hex.parse(hex);
164 		this.md.update(wHex);
165 	    };
166 	    this.digest = function() {
167 		var hash = this.md.finalize();
168 		return hash.toString(CryptoJS.enc.Hex);
169 	    };
170 	    this.digestString = function(str) {
171 		this.updateString(str);
172 		return this.digest();
173 	    };
174 	    this.digestHex = function(hex) {
175 		this.updateHex(hex);
176 		return this.digest();
177 	    };
178 	}
179     };
180 
181     /**
182      * update digest by specified string
183      * @name updateString
184      * @memberOf KJUR.crypto.MessageDigest
185      * @function
186      * @param {String} str string to update
187      * @description
188      * @example
189      * md.updateString('New York');
190      */
191     this.updateString = function(str) {
192 	throw "updateString(str) not supported for this alg/prov: " + this.algName + "/" + this.provName;
193     };
194 
195     /**
196      * update digest by specified hexadecimal string
197      * @name updateHex
198      * @memberOf KJUR.crypto.MessageDigest
199      * @function
200      * @param {String} hex hexadecimal string to update
201      * @description
202      * @example
203      * md.updateHex('0afe36');
204      */
205     this.updateHex = function(hex) {
206 	throw "updateHex(hex) not supported for this alg/prov: " + this.algName + "/" + this.provName;
207     };
208 
209     /**
210      * completes hash calculation and returns hash result
211      * @name digest
212      * @memberOf KJUR.crypto.MessageDigest
213      * @function
214      * @description
215      * @example
216      * md.digest()
217      */
218     this.digest = function() {
219 	throw "digest() not supported for this alg/prov: " + this.algName + "/" + this.provName;
220     };
221 
222     /**
223      * performs final update on the digest using string, then completes the digest computation
224      * @name digestString
225      * @memberOf KJUR.crypto.MessageDigest
226      * @function
227      * @param {String} str string to final update
228      * @description
229      * @example
230      * md.digestString('aaa')
231      */
232     this.digestString = function(str) {
233 	throw "digestString(str) not supported for this alg/prov: " + this.algName + "/" + this.provName;
234     };
235 
236     /**
237      * performs final update on the digest using hexadecimal string, then completes the digest computation
238      * @name digestHex
239      * @memberOf KJUR.crypto.MessageDigest
240      * @function
241      * @param {String} hex hexadecimal string to final update
242      * @description
243      * @example
244      * md.digestHex('0f2abd')
245      */
246     this.digestHex = function(hex) {
247 	throw "digestHex(hex) not supported for this alg/prov: " + this.algName + "/" + this.provName;
248     };
249 
250     if (typeof params != "undefined") {
251 	if (typeof params['alg'] != "undefined") {
252 	    this.algName = params['alg'];
253 	    this.provName = params['prov'];
254 	    this.setAlgAndProvider(params['alg'], params['prov']);
255 	}
256     }
257 };

260 /**
261  * Signature class which is very similar to java.security.Signature class
262  * @name KJUR.crypto.Signature
263  * @class Signature class which is very similar to java.security.Signature class
264  * @param {Array} params parameters for constructor
265  * @description
266  * <br/>
267  * As for params of constructor's argument, it can be specify following attributes:
268  * <ul>
269  * <li>alg - signature algorithm name (ex. {MD5,SHA1,SHA224,SHA256,SHA384,SHA512,RIPEMD160}withRSA)</li>
270  * <li>provider - currently 'cryptojs/jsrsa' only</li>
271  * <li>prvkeypem - PEM string of signer's private key. If this specified, no need to call initSign(prvKey).</li>
272  * </ul>
273  * <h4>SUPPORTED ALGORITHMS AND PROVIDERS</h4>
274  * Signature class now supports {MD5,SHA1,SHA224,SHA256,SHA384,SHA512,RIPEMD160}
275  * withRSA algorithm in 'cryptojs/jsrsa' provider.
276  * <h4>EXAMPLES</h4>
277  * @example
278  * var sig = new KJUR.crypto.Signature({"alg": "SHA1withRSA", "prov": "cryptojs/jsrsa"});
279  * // initialize
280  * sig.initSign(prvKey)
281  * sig.initVerifyByCert(cert) // not yet supported
282  * sig.initVerifyByPubKey(pubKey)
283  * // append data
284  * sig.updateHex(hex)
285  * sig.updateString(str)
286  * // get signature
287  * sig.sign()
288  * sig.signHex(hex)
289  * sig.signString(str)
290  * // verify
291  * verifyHex(sigValHex) // not yet supported
292  */
293 KJUR.crypto.Signature = function(params) {
294     var prvKey = null;
295     var md = null; // KJUR.crypto.MessageDigest object
296     var sig = null;
297     var algName = null;
298     var provName = null;
299     var algProvName = null;
300     var mdAlgName = null;
301     var pubkeyAlgName = null;
302     var mode = null;
303 
304     var sHashHex = null; // hex hash value for hex
305     var hDigestInfo = null;
306     var hPaddedDigestInfo = null;
307     var hSign = null;
308 
309     this._setAlgNames = function() {
310 	if (this.algName.match(/^(.+)with(.+)$/)) {
311 	    this.mdAlgName = RegExp.$1.toLowerCase();
312 	    this.pubkeyAlgName = RegExp.$2.toLowerCase();
313 	}
314     };
315 
316     this._zeroPaddingOfSignature = function(hex, bitLength) {
317 	var s = "";
318 	var nZero = bitLength / 4 - hex.length;
319 	for (var i = 0; i < nZero; i++) {
320 	    s = s + "0";
321 	}
322 	return s + hex;
323     };
324 
325     /**
326      * set signature algorithm and provider
327      * @name setAlgAndProvider
328      * @memberOf KJUR.crypto.Signature
329      * @function
330      * @param {String} alg signature algorithm name
331      * @param {String} prov provider name
332      * @description
333      * <br/>
334      * <h4>EXAMPLES</h4>
335      * @example
336      * md.setAlgAndProvider('SHA1withRSA', 'cryptojs/jsrsa');
337      */
338     this.setAlgAndProvider = function(alg, prov) {
339 	this._setAlgNames();
340 	if (prov != 'cryptojs/jsrsa')
341 	    throw "provider not supported: " + prov;
342 
343 	if (':md5:sha1:sha224:sha256:sha384:sha512:ripemd160:'.indexOf(this.mdAlgName) != -1) {
344 	    try {
345 		this.md = new KJUR.crypto.MessageDigest({'alg':this.mdAlgName,'prov':'cryptojs'});
346 	    } catch (ex) {
347 		throw "setAlgAndProvider hash alg set fail alg=" + this.mdAlgName + "/" + ex;
348 	    }
349 
350 	    this.initSign = function(prvKey) {
351 		this.prvKey = prvKey;
352 	    };
353 
354 	    this.updateString = function(str) {
355 		this.md.updateString(str);
356 	    };
357 	    this.updateHex = function(hex) {
358 		this.md.updateHex(hex);
359 	    };
360 	    this.sign = function() {
361                 var util = KJUR.crypto.Util;
362 		var keyLen = this.prvKey.n.bitLength();
363 		this.sHashHex = this.md.digest();
364 		this.hDigestInfo = util.getDigestInfoHex(this.sHashHex, this.mdAlgName);
365 		this.hPaddedDigestInfo = 
366                     util.getPaddedDigestInfoHex(this.sHashHex, this.mdAlgName, keyLen);
367 		var biPaddedDigestInfo = parseBigInt(this.hPaddedDigestInfo, 16);
368 		this.hoge = biPaddedDigestInfo.toString(16);
369 		var biSign = this.prvKey.doPrivate(biPaddedDigestInfo);
370 		this.hSign = this._zeroPaddingOfSignature(biSign.toString(16), keyLen);
371 		return this.hSign;
372 	    };
373 	    this.signString = function(str) {
374 		this.updateString(str);
375 		this.sign();
376 	    };
377 	    this.signHex = function(hex) {
378 		this.updateHex(hex);
379 		this.sign();
380 	    };
381 	}
382     };
383 
384     /**
385      * Initialize this object for signing
386      * @name initSign
387      * @memberOf KJUR.crypto.Signature
388      * @function
389      * @param {RSAKey} prvKey RSAKey object of private key
390      * @description
391      * @example
392      * sig.initSign(prvKey)
393      */
394     this.initSign = function(prvKey) {
395 	throw "initSign(prvKey) not supported for this alg:prov=" + this.algProvName;
396     };
397 
398     /**
399      * Updates the data to be signed or verified by a string
400      * @name updateString
401      * @memberOf KJUR.crypto.Signature
402      * @function
403      * @param {String} str string to use for the update
404      * @description
405      * @example
406      * sig.updateString('aaa')
407      */
408     this.updateString = function(str) {
409 	throw "updateString(str) not supported for this alg:prov=" + this.algProvName;
410     };
411 
412     /**
413      * Updates the data to be signed or verified by a hexadecimal string
414      * @name updateHex
415      * @memberOf KJUR.crypto.Signature
416      * @function
417      * @param {String} hex hexadecimal string to use for the update
418      * @description
419      * @example
420      * sig.updateHex('1f2f3f')
421      */
422     this.updateHex = function(hex) {
423 	throw "updateHex(hex) not supported for this alg:prov=" + this.algProvName;
424     };
425 
426     this.signString = function(str) {
427 	throw "digestString(str) not supported for this alg:prov=" + this.algProvName;
428     };
429 
430     this.signHex = function(hex) {
431 	throw "digestHex(hex) not supported for this alg:prov=" + this.algProvName;
432     };
433 
434     /**
435      * Returns the signature bytes of all data updates as a hexadecimal string
436      * @name sign
437      * @memberOf KJUR.crypto.Signature
438      * @function
439      * @return the signature bytes as a hexadecimal string
440      * @description
441      * @example
442      * var hSigValue = sig.sign()
443      */
444     this.sign = function() {
445 	throw "sign() not supported for this alg:prov=" + this.algProvName;
446     };
447 
448     if (typeof params != "undefined") {
449 	if (typeof params['alg'] != "undefined") {
450 	    this.algName = params['alg'];
451 	    this.provName = params['prov'];
452 	    this.algProvName = params['alg'] + ":" + params['prov'];
453 	    this.setAlgAndProvider(params['alg'], params['prov']);
454 	    this._setAlgNames();
455 	}
456 	if (typeof params['prvkeypem'] != "undefined") {
457 	    if (typeof params['prvkeypas'] != "undefined") {
458 		throw "both prvkeypem and prvkeypas parameters not supported";
459 	    } else {
460 		try {
461 		    var prvKey = new RSAKey();
462 		    prvKey.readPrivateKeyFromPEMString(params['prvkeypem']);
463 		    this.initSign(prvKey);
464 		} catch (ex) {
465 		    throw "fatal error to load pem private key: " + ex;
466 		}
467 	    }
468 	}
469     }
470 };
