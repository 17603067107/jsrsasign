 17 /*
 18  * MEMO:
 19  *   f('3082025b02...', 2) ... 82025b ... 3bytes
 20  *   f('020100', 2) ... 01 ... 1byte
 21  *   f('0203001...', 2) ... 03 ... 1byte
 22  *   f('02818003...', 2) ... 8180 ... 2bytes
 23  *   f('3080....0000', 2) ... 80 ... -1
 24  *
 25  *   Requirements:
 26  *   - ASN.1 type octet length MUST be 1. 
 27  *     (i.e. ASN.1 primitives like SET, SEQUENCE, INTEGER, OCTETSTRING ...)
 28  */
 29

 38 /**
 39  * ASN.1 DER encoded hexadecimal string utility class
 40  * @name ASN1HEX
 41  * @class ASN.1 DER encoded hexadecimal string utility class
 42  * @since jsrsasign 1.1
 43  */
 44 var ASN1HEX = new function() {
 45     /**
 46      * get byte length for ASN.1 L(length) bytes
 47      * @name getByteLengthOfL_AtObj
 48      * @memberOf ASN1HEX
 49      * @function
 50      * @param {String} s hexadecimal string of ASN.1 DER encoded data
 51      * @param {Number} pos string index
 52      * @return byte length for ASN.1 L(length) bytes
 53      */
 54     this.getByteLengthOfL_AtObj = function(s, pos) {
 55 	if (s.substring(pos + 2, pos + 3) != '8') return 1;
 56 	var i = parseInt(s.substring(pos + 3, pos + 4));
 57 	if (i == 0) return -1; 		// length octet '80' indefinite length
 58 	if (0 < i && i < 10) return i + 1;	// including '8?' octet;
 59 	return -2;				// malformed format
 60     };
 61 
 62     /**
 63      * get hexadecimal string for ASN.1 L(length) bytes
 64      * @name getHexOfL_AtObj
 65      * @memberOf ASN1HEX
 66      * @function
 67      * @param {String} s hexadecimal string of ASN.1 DER encoded data
 68      * @param {Number} pos string index
 69      * @return {String} hexadecimal string for ASN.1 L(length) bytes
 70      */
 71     this.getHexOfL_AtObj = function(s, pos) {
 72 	var len = this.getByteLengthOfL_AtObj(s, pos);
 73 	if (len < 1) return '';
 74 	return s.substring(pos + 2, pos + 2 + len * 2);
 75     };
 76 
 77     //   getting ASN.1 length value at the position 'idx' of
 78     //   hexa decimal string 's'.
 79     //
 80     //   f('3082025b02...', 0) ... 82025b ... ???
 81     //   f('020100', 0) ... 01 ... 1
 82     //   f('0203001...', 0) ... 03 ... 3
 83     //   f('02818003...', 0) ... 8180 ... 128
 84     /**
 85      * get integer value of ASN.1 length for ASN.1 data
 86      * @name getIntOfL_AtObj
 87      * @memberOf ASN1HEX
 88      * @function
 89      * @param {String} s hexadecimal string of ASN.1 DER encoded data
 90      * @param {Number} pos string index
 91      * @return ASN.1 L(length) integer value
 92      */
 93     this.getIntOfL_AtObj = function(s, pos) {
 94 	var hLength = this.getHexOfL_AtObj(s, pos);
 95 	if (hLength == '') return -1;
 96 	var bi;
 97 	if (parseInt(hLength.substring(0, 1)) < 8) {
 98 	    bi = new BigInteger(hLength, 16);
 99 	} else {
100 	    bi = new BigInteger(hLength.substring(2), 16);
101 	}
102 	return bi.intValue();
103     };
104 
105     /**
106      * get ASN.1 value starting string position for ASN.1 object refered by index 'idx'.
107      * @name getStartPosOfV_AtObj
108      * @memberOf ASN1HEX
109      * @function
110      * @param {String} s hexadecimal string of ASN.1 DER encoded data
111      * @param {Number} pos string index
112      */
113     this.getStartPosOfV_AtObj = function(s, pos) {
114 	var l_len = this.getByteLengthOfL_AtObj(s, pos);
115 	if (l_len < 0) return l_len;
116 	return pos + (l_len + 1) * 2;
117     };
118 
119     /**
120      * get hexadecimal string of ASN.1 V(value)
121      * @name getHexOfV_AtObj
122      * @memberOf ASN1HEX
123      * @function
124      * @param {String} s hexadecimal string of ASN.1 DER encoded data
125      * @param {Number} pos string index
126      * @return {String} hexadecimal string of ASN.1 value.
127      */
128     this.getHexOfV_AtObj = function(s, pos) {
129 	var pos1 = this.getStartPosOfV_AtObj(s, pos);
130 	var len = this.getIntOfL_AtObj(s, pos);
131 	return s.substring(pos1, pos1 + len * 2);
132     };
133 
134     /**
135      * get hexadecimal string of ASN.1 TLV at
136      * @name getHexOfTLV_AtObj
137      * @memberOf ASN1HEX
138      * @function
139      * @param {String} s hexadecimal string of ASN.1 DER encoded data
140      * @param {Number} pos string index
141      * @return {String} hexadecimal string of ASN.1 TLV.
142      * @since 1.1
143      */
144     this.getHexOfTLV_AtObj = function(s, pos) {
145 	var hT = s.substr(pos, 2);
146 	var hL = this.getHexOfL_AtObj(s, pos);
147 	var hV = this.getHexOfV_AtObj(s, pos);
148 	return hT + hL + hV;
149     };
150 
151     /**
152      * get next sibling starting index for ASN.1 object string
153      * @name getPosOfNextSibling_AtObj
154      * @memberOf ASN1HEX
155      * @function
156      * @param {String} s hexadecimal string of ASN.1 DER encoded data
157      * @param {Number} pos string index
158      * @return next sibling starting index for ASN.1 object string
159      */
160     this.getPosOfNextSibling_AtObj = function(s, pos) {
161 	var pos1 = this.getStartPosOfV_AtObj(s, pos);
162 	var len = this.getIntOfL_AtObj(s, pos);
163 	return pos1 + len * 2;
164     };
165 
166     /**
167      * get array of indexes of child ASN.1 objects
168      * @name getPosArrayOfChildren_AtObj
169      * @memberOf ASN1HEX
170      * @function
171      * @param {String} s hexadecimal string of ASN.1 DER encoded data
172      * @param {Number} start string index of ASN.1 object
173      * @return {Array of Number} array of indexes for childen of ASN.1 objects
174      */
175     this.getPosArrayOfChildren_AtObj = function(h, pos) {
176 	var a = new Array();
177 	var p0 = this.getStartPosOfV_AtObj(h, pos);
178 	a.push(p0);
179 
180 	var len = this.getIntOfL_AtObj(h, pos);
181 	var p = p0;
182 	var k = 0;
183 	while (1) {
184 	    var pNext = this.getPosOfNextSibling_AtObj(h, p);
185 	    if (pNext == null || (pNext - p0  >= (len * 2))) break;
186 	    if (k >= 200) break;
187 	    
188 	    a.push(pNext);
189 	    p = pNext;
190 	    
191 	    k++;
192 	}
193 	
194 	return a;
195     };
196 
197     /**
198      * get string index of nth child object of ASN.1 object refered by h, idx
199      * @name getNthChildIndex_AtObj
200      * @memberOf ASN1HEX
201      * @function
202      * @param {String} h hexadecimal string of ASN.1 DER encoded data
203      * @param {Number} idx start string index of ASN.1 object
204      * @param {Number} nth for child
205      * @return {Number} string index of nth child.
206      * @since 1.1
207      */
208     this.getNthChildIndex_AtObj = function(h, idx, nth) {
209 	var a = this.getPosArrayOfChildren_AtObj(h, idx);
210 	return a[nth];
211     };
212 
213     // ========== decendant methods ==============================
214     /**
215      * get string index of nth child object of ASN.1 object refered by h, idx
216      * @name getDecendantIndexByNthList
217      * @memberOf ASN1HEX
218      * @function
219      * @param {String} h hexadecimal string of ASN.1 DER encoded data
220      * @param {Number} currentIndex start string index of ASN.1 object
221      * @param {Array of Number} nthList array list of nth
222      * @return {Number} string index refered by nthList
223      * @since 1.1
224      */
225     this.getDecendantIndexByNthList = function(h, currentIndex, nthList) {
226 	if (nthList.length == 0) {
227 	    return currentIndex;
228 	}
229 	var firstNth = nthList.shift();
230 	var a = this.getPosArrayOfChildren_AtObj(h, currentIndex);
231 	return this.getDecendantIndexByNthList(h, a[firstNth], nthList);
232     };
233 
234     /**
235      * get hexadecimal string of ASN.1 TLV refered by current index and nth index list.
236      * @name getDecendantHexTLVByNthList
237      * @memberOf ASN1HEX
238      * @function
239      * @param {String} h hexadecimal string of ASN.1 DER encoded data
240      * @param {Number} currentIndex start string index of ASN.1 object
241      * @param {Array of Number} nthList array list of nth
242      * @return {Number} hexadecimal string of ASN.1 TLV refered by nthList
243      * @since 1.1
244      */
245     this.getDecendantHexTLVByNthList = function(h, currentIndex, nthList) {
246 	var idx = this.getDecendantIndexByNthList(h, currentIndex, nthList);
247 	return this.getHexOfTLV_AtObj(h, idx);
248     };
249 
250     /**
251      * get hexadecimal string of ASN.1 V refered by current index and nth index list.
252      * @name getDecendantHexVByNthList
253      * @memberOf ASN1HEX
254      * @function
255      * @param {String} h hexadecimal string of ASN.1 DER encoded data
256      * @param {Number} currentIndex start string index of ASN.1 object
257      * @param {Array of Number} nthList array list of nth
258      * @return {Number} hexadecimal string of ASN.1 V refered by nthList
259      * @since 1.1
260      */
261     this.getDecendantHexVByNthList = function(h, currentIndex, nthList) {
262 	var idx = this.getDecendantIndexByNthList(h, currentIndex, nthList);
263 	return this.getHexOfV_AtObj(h, idx);
264     };
265 };
